```c++
{

		cin>>n>>c;
```

### [1730B - Meeting on the Line](https://codeforces.com/contest/1730/problem/B)

```c++
#include<bits/stdc++.h>
using namespace std;
#define int long long 
#define endl "\n"
const int INF=0x7fffffff;
const int maxn=1e7+10;
int a[maxn],t[maxn];
int ti,n,ma=-1,mi=INF,mai=-1,mat=-1,di1,di2;
signed main()
{
	//ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
	cin>>ti;
	while(ti--){
		mai=-1; mat=-1; mi=INF; ma=-1;
		int flag=0;
		cin>>n;
		for(int i=1;i<=n;i++){
			cin>>a[i];
			if(i!=1&&a[i]!=a[i-1]){
				flag=1;
			}
			mi=min(mi,a[i]);
			ma=max(ma,a[i]);
		}
		for(int i=1;i<=n;i++){
			cin>>t[i];
			if(t[i]>mat){
				mai=a[i];
				mat=t[i];
			}
		}
		if(flag==0){
			double ans=double(mi)+double(ma-mi)/2;
			printf("%.6lf\n",ans);
			continue;
		}
		mi=INF; ma=-1;
		for(int i=1;i<=n;i++){
			di1=mat-t[i];
			di2=abs(mai-a[i]);
			if(mai==a[i]) continue;
			else if(mai>a[i]){
				if(di1>=di2){
					a[i]=mai;
				}
				else a[i]+=di1;
			}
			else{
				if(di1>=di2){
					a[i]=mai;
				}
				else a[i]-=di1;
			}
		}
		for(int i=1;i<=n;i++){
			mi=min(mi,a[i]);
			ma=max(ma,a[i]);
		}
		double ans=double(mi)+double(ma-mi)/2;
		printf("%.6lf\n",ans);
	}
	return 0;
}
```

### [1730C - Minimum Notation](https://codeforces.com/contest/1730/problem/C)



```c++
#include<bits/stdc++.h>
using namespace std;
#define int long long 
#define endl "\n"
int n,t;
void solve(){
	string s;
	cin>>s;
	n=s.size();
	vector<int> a(n); //数字数组 
	vector<int> mn(n+1,9); //后缀最小数组 
	for(int i=0;i<n;i++){
		a[i]=s[i]-'0';
	}
	for(int i=n-1;i>=0;i--){
		mn[i]=min(mn[i+1],a[i]);  //记录每个点后缀最小值 
	}
	vector<int> buff(10,0); //0~10初始化为0 
	string ans="";
	string t="0123456789";
	for(int i=0;i<n;i++){
		for(int j=0;j<mn[i];j++){
			while(buff[j]){
				buff[j]--;
				ans+=t[j];
			} 
		}
		if(a[i]==mn[i]){ //如果此值等于后缀最小值 
			ans+=t[a[i]];
		}
		else{
			buff[min((int)9,a[i]+1)]++; 
			//如果此值大于后缀最小值，则添加到buff,之后添加到ans尾部 
		}
	}
	for(int j=0;j<10;j++){ //按照字典序添加 
		while(buff[j]){
			buff[j]--;
			ans+=t[j];
		}
	}
	cout<<ans<<endl;
}
signed main()
{
	ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
	cin>>t;
	while(t--){
		solve();
	}
	return 0;
}
```
