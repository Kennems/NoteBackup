1.除法会失真，乘法转化为除法 

2.考虑正负数，负数乘一个正数会变小

3.保证全部输入再输出

```c++
#include<bits/stdc++.h>
using namespace std;
#define int long long 
#define endl "\n"
const int maxn=1e7+10; 
int t,n,k,flag;
int s[maxn],a[maxn];
signed main()
{
	ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
	cin>>t;
	while(t--){
		cin>>n>>k;
		flag=0;
		for(int i=1;i<=k;i++){
			cin>>s[i];
			if(i!=1){
				a[i]=s[i]-s[i-1];
			}
			else if(i==1){
				a[i]=s[i];
			}
		}
		if(k==1){
			cout<<"Yes"<<endl;
			continue;
		}
		if(a[1]>a[2]){
			if(a[1]>a[2]*(n-k+1)) flag=1;//除法化为乘法
		}
		if(a[1]>a[2]*(n-k+1)) flag=1; 
		for(int i=2;i<k;i++){
			if(a[i]>a[i+1]){
				flag=1;
				break;
			}
		}
		if(flag){
			cout<<"No"<<endl;
		}
		else{
			cout<<"Yes"<<endl;
		}
	}
	return 0;
}
```
