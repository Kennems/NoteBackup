/**
 * struct ListNode {
 *  int val;
 *  struct ListNode *next;
 *  ListNode(int x) : val(x), next(nullptr) {}
 * };
 */
class Solution {
  public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     *
     * @param l1 ListNode类
     * @param l2 ListNode类
     * @return ListNode类
     */
    ListNode* merge_cows(ListNode* l1, ListNode* l2) {
        // write code here
        ListNode* head = new ListNode(0);
        ListNode* nxt = head;
        while (l1 && l2) {
            if (l1->val < l2->val) {
                nxt ->next = l1 ;
                l1 = l1->next;
            } else {
                nxt->next = l2;
                l2 = l2 ->next;
            }
            nxt = nxt -> next;
        }
        nxt ->next = l1 == nullptr ? l2 : l1;
        return head ->next;
    }
};
