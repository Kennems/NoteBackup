## D

先确定min(p,q)小于根三N,因为如果p,q均大于根三N则结果一定大于n.。

所以可得

```c++
#include<bits/stdc++.h>
#define int long long 
#define endl "\n"
using namespace std;
int n,t,p,q;
signed main()
{
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    cin>>t;
    while(t--){
        cin>>n;
        for(int i=2;i*i*i<=n;i++){
            if(n%i!=0) continue;
            if((n/i)%i==0){
                p=i; q=(n/i)/i;
            }
            else{
                q=i;
                p=sqrt(n/i);
            }
            break;
        }
        cout<<p<<" "<<q<<endl;
    }
    return 0;
}
```
