import sys
input = lambda:sys.stdin.readline().strip()
MOD = 998244353

class SegmentTree:
    def __init__(self, arr):
        self.size = len(arr)
        self.tree = [0] * (4 * self.size)
        self.lazy = [0] * (4 * self.size)
        self.init_tree(arr, 0, 0, self.size - 1)
        
    def init_tree(self, arr, node, start, end):
        if start == end:
            self.tree[node] = arr[start]
        else:
            mid = (start + end) // 2
            self.init_tree(arr, 2 * node + 1, start, mid)
            self.init_tree(arr, 2 * node + 2, mid + 1, end)
            self.tree[node] = (self.tree[2 * node + 1] * self.tree[2 * node + 2]) % MOD
    
    def update(self, node, start, end, left, right, x):
        if self.lazy[node] != 0:
            self.tree[node] = pow(self.lazy[node], end - start + 1, MOD)
            if start != end:
                self.lazy[2 * node + 1] = self.lazy[node]
                self.lazy[2 * node + 2] = self.lazy[node]
            self.lazy[node] = 0
        
        if start > right or end < left:
            return
        
        if left <= start and end <= right:
            self.tree[node] = pow(x, end - start + 1, MOD)
            if start != end:
                self.lazy[2 * node + 1] = x
                self.lazy[2 * node + 2] = x
            return
        
        mid = (start + end) // 2
        self.update(2 * node + 1, start, mid, left, right, x)
        self.update(2 * node + 2, mid + 1, end, left, right, x)
        self.tree[node] = (self.tree[2 * node + 1] * self.tree[2 * node + 2]) % MOD
    
    def query(self, node, start, end, left, right):
        if self.lazy[node] != 0:
            self.tree[node] = pow(self.lazy[node], end - start + 1, MOD)
            if start != end:
                self.lazy[2 * node + 1] = self.lazy[node]
                self.lazy[2 * node + 2] = self.lazy[node]
            self.lazy[node] = 0
        
        if start > right or end < left:
            return 1
        
        if left <= start and end <= right:
            return self.tree[node]
        
        mid = (start + end) // 2
        left_prod = self.query(2 * node + 1, start, mid, left, right)
        right_prod = self.query(2 * node + 2, mid + 1, end, left, right)
        return (left_prod * right_prod) % MOD

n, m = map(int, input().split())
snowflakes = list(map(int, input().split()))
tree = SegmentTree(snowflakes)

for _ in range(m):
    op = list(map(int, input().split()))
    if op[0] == 1:
        tree.update(0, 0, n - 1, op[1] - 1, op[2] - 1, op[3])
    else:
        print(tree.query(0, 0, n - 1, op[1] - 1, op[2] - 1))
